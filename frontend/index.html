④ Firmware ESP32 (Arduino IDE)
Bibliothèques : TinyGPS++, HTTPClient, WiFi
collier_geoalerte.ino 
// ── Bibliothèques ──────────────────────
#include <WiFi.h>
#include <HTTPClient.h>
#include <TinyGPS++.h>

// ── À personnaliser ────────────────────
const char* WIFI_SSID   = "VOTRE_WIFI";
const char* WIFI_PASS   = "MOT_DE_PASSE";
const char* SERVER_URL  = "https://dawbali-backend.onrender.com/api/iot/gps";
const char* API_KEY     = "geoalerte_iot_key_2024";
const int   ID_TROUPEAU = 1;        // ID dans la BD
const long  INTERVAL_MS = 60000;   // 1 envoi / minute

// ── GPS sur Serial2 (RX=16, TX=17) ────
TinyGPSPlus gps;
HardwareSerial gpsSerial(2);

void setup() {
  Serial.begin(115200);
  gpsSerial.begin(9600, SERIAL_8N1, 16, 17);
  WiFi.begin(WIFI_SSID, WIFI_PASS);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500); Serial.print(".");
  }
  Serial.println("\nWiFi OK");
}

unsigned long lastSend = 0;

void loop() {
  while (gpsSerial.available())
    gps.encode(gpsSerial.read());

  if (millis() - lastSend >= INTERVAL_MS && gps.location.isValid()) {
    sendPosition(gps.location.lng(), gps.location.lat(),
                 gps.hdop.value() / 100.0,
                 gps.course.deg(), readBattery());
    lastSend = millis();
  }
}

void sendPosition(double lon, double lat,
                   float hdop, double cap, int bat) {
  HTTPClient http;
  http.begin(SERVER_URL);
  http.addHeader("Content-Type", "application/json");
  String body =
    "{\"api_key\":\""      + String(API_KEY)      + "\","
    "\"id_troupeau\":"    + ID_TROUPEAU          + ","
    "\"longitude\":"      + String(lon, 6)      + ","
    "\"latitude\":"       + String(lat, 6)      + ","
    "\"precision_pos\":"  + String(hdop, 1)     + ","
    "\"direction\":"      + String(cap, 0)      + ","
    "\"niveau_batterie\":"+ bat                  + "}";
  int code = http.POST(body);
  http.end();
  Serial.println(code == 200 ? "✓ OK" : "✗ " + String(code));
}

int readBattery() {
  // Diviseur résistif 100kΩ/47kΩ sur GPIO34
  float v = analogRead(34) * 3.3f / 4095.0f * 3.14f;
  return constrain(int((v - 3.0f) / 1.2f * 100), 0, 100);
}

Conseils mise en production
🔑
Sécurité clé API
Changer API_KEY_IOT dans Render → Environment. Si compromise, changer uniquement côté serveur.
⏱
Fréquence d'envoi
Transhumance : 5 min · Pâturage : 30 min · Repos : 1 h. Ajuster INTERVAL_MS dans le firmware.
📶
Zones sans réseau
Stocker les trames en flash (LittleFS) et transmettre par lot au retour réseau.
✅
Vérifier l'intégration
Après le premier envoi → page Carte en direct : le marqueur GPS doit apparaître. Toute entrée dans une zone génère une alerte SSE instantanée.
